196
197m_value_type DynamicVariableSequence::getValue (m_time_type time,
198 m_time_type timeTotal)
199{
200 if (time > timeTotal || time < 0)
201 {
202 // they are dumb for passing in invalid values, so let them know
203 cout << "DVS::getValue(...): invalid time or totalTime parameter" << endl;
204 }
205
206 //Handle special case where absolute end of dvs is asked for
207 if(time==timeTotal)
208 {
209 return xyPoints_->get(xyPoints_->size()-1).y;
210 }
211
212 //TODO: Make come sort of cached flag that will see if this
213 // actually needs to be performed or not.
214 generateTimes(timeTotal);
215
216 //First, find the segment index that this time/timeTotal refers to
217 m_time_type current=0;
218 int timeIndex=0;
219 while(current<time)
220 {
221 current+=generatedSegmentTimes_->get(timeIndex);
222 timeIndex++;
223 }
224
225 //If we have overshot, lets go back
226 if(current!=time)
227 {
228 timeIndex--;
229 current-=generatedSegmentTimes_->get(timeIndex);
230 }
231
232 //Spawn an interpolator of the proper type
233 Interpolator *interp;
234 switch (getSegmentInterpolationType(timeIndex))
235 {
236 case EXPONENTIAL:
237 interp=new ExponentialInterpolator();
238 break;
239
240 case CUBIC_SPLINE:
241 interp=new CubicSplineInterpolator();
242 break;
243
244 default:
245 interp=new LinearInterpolator();
246 break;
247 }
248
249 // set the Interpolator's sampling rate to a fixed granularity
250 interp->setSamplingRate(100);
251
252 // set interpolator duration to one second to make it easy
253 interp->setDuration(1.0);
254
255 // We know that this segment interpolates between xy point
256 // timeIndex and timeIndex+1
257 xy_point left,right;
258 left=getPoint(timeIndex);
259 right=getPoint(timeIndex+1);
260
261 // Now tell the interpolator what values we are going between
262 interp->addEntry(0,left.y);
263 interp->addEntry(1,right.y);
264
265 // Figure out which sample we want from the interpolator.
266 // This is done by taking the percentage we have to iterate thru
267 // the interpolator, and then multiplying that by the number of samples
268 int sample=int(round(((time-current)/generatedSegmentTimes_->get(timeIndex))*100.0));
269
270 // Create a value iterator to get values from the interpolator
271 Iterator<m_value_type> tempIterator=interp->valueIterator();
272
273 // Now lets skip the first sample-1 samples...
274 for(int i=0;i<sample;i++)
275 {
276 tempIterator.next();
277 }
278
279 // Now grab the one we want
280 m_value_type returnval=tempIterator.next();
281
282 delete interp;
283
284 return returnval;
285}

346inline void DynamicVariableSequence::DefineShape
347 (Collection<xy_point> xyPoints,
348 Collection<envelope_segment> segments)
349{
350 // make sure we don't cause a memory leak
351 if (xyPoints_ != NULL)
352 {
353 delete (xyPoints_);
354 }
355 if (segments_ != NULL)
356 {
357 delete (segments_);
358 }
359
360 if (xyPoints.size ( ) != (segments.size ( ) + 1) )
361 {
362 cout << "WARNING: Invalid Collection sizes for points or segments" << endl;
363 cout << "points.size ( ) should be segments.size ( ) + 1" << endl;
364 }
365
366 // set our local variables to copies of the variables passed in
367 xyPoints_ = new Collection<xy_point> (xyPoints);
368 segments_ = new Collection<envelope_segment> (segments);
369}

371//----------------------------------------------------------------------------//
372void DynamicVariableSequence::AddToShape
373(Collection<xy_point> xyPoints,
374 Collection<envelope_segment> segments)
375{
376 // case: Adding to an empty Collection
377 if ( (xyPoints_ == NULL) || (segments_ == NULL) )
378 {
379 xyPoints_ = new Collection <xy_point> (xyPoints);
380 segments_ = new Collection <envelope_segment> (segments);
381 return;
382 }
383
384 int iNumSegments = segments.size();
385 m_time_type timeOffset = xyPoints_->get(xyPoints_->size() - 1).x;
386 xy_point pointTemp;
387
388 // verify that the last old point and the first new point have the
389 // same y value (usually zero), that they can be appended
390 if ( (xyPoints_ -> get (xyPoints_ -> size ( ) -1).y)
391 != (xyPoints.get (0).y ) )
392 {
393 cout << "WARNING: Y values of last point of first envelope\n";
394 cout << "and first point of second envelope do not match.\n";
395 cout << "This implementation corrupts the first point of the\n";
396 cout << "second envelope\n";
397 }
398
399 // for every element in the new collections
400 for (int iIndex = 0; iIndex < iNumSegments; iIndex++)
401 {
402 // add the current segment and point + 1 to our existing Collections
403 pointTemp = xyPoints.get(iIndex + 1);
404 pointTemp.x = pointTemp.x + timeOffset;
405 xyPoints_->add(pointTemp);
406 segments_->add(segments.get(iIndex));
407 }
408}

563//----------------------------------------------------------------------------//
564Iterator<m_value_type> DynamicVariableSequence::valueIterator()
565{
566 // optimize so that if the envelope is used repeatedly without changing
567 // the time value that is assigned, interpolators are not re-created
568 // every time
569 if (totalTime_ != this->getDuration()
570 || currentInterpolatorRate_ != this->getSamplingRate())
571 {
572 totalTime_ = this->getDuration();
573 currentInterpolatorRate_ = this->getSamplingRate();
574
575 // populate the times member variable with times
576 generateTimes(totalTime_);
577
578 // populate the interpolator member variable with interpolators, based
579 // on the times just set
580 addInterpolators(currentInterpolatorRate_);
581 }
582
583 // give them the big iterator
584 return (Iterator<m_value_type>
585 (new DynamicVariableSequenceIterator(*interpolators_)));
586}

589//----------------------------------------------------------------------------//
590void DynamicVariableSequence::generateTimes (m_time_type totalTime)
591{
592#ifdef DEBUG_MODE
593 cout << "DVS::generateTimes" << endl;
594#endif
595 int iNumSegments = segments_->size();
596 double dTotalFixedTime = 0;
597 double dTotalFlexPercent = 0;
598 double dScaleRatio = 1;
599
600 totalTime_ = totalTime;
601
602 // get rid of previously generated times and make a new collection
603 if (generatedSegmentTimes_ != NULL)
604 {
605 delete (generatedSegmentTimes_);
606 }
607 generatedSegmentTimes_ = new Collection<m_time_type> ();
608
609 // figure out how they want to allocate time for each segment and populate
610 // the generated times with their current values
611 for (int iIndex = 0; iIndex < iNumSegments; iIndex++)
612 {
613 // if this segment has fixed-time, increment fixed time running total
614 if (this->getSegmentTimeType(iIndex) == FIXED)
615 {
616 dTotalFixedTime += this->getSegmentTime(iIndex);
617 }
618 // else increment flex-time running total
619 else
620 {
621 dTotalFlexPercent += this->getSegmentTime(iIndex);
622 }
623 // append their current value to our values for generated times
624 generatedSegmentTimes_->add(this->getSegmentTime(iIndex));
625 }
626
627 // now, correct the generated values as necessary (i.e. the rest of this)
628
629 // if they specified more FIXED time than actual time, or if there is no
630 // FIEXIBLE time and they did not specify enough FIXED time, scale the
631 // FIXED times to the actual time (and set total amount of FLEXIBLE time
632 // to zero)
633 if (dTotalFixedTime > totalTime_
634 || (dTotalFixedTime < totalTime_ && dTotalFlexPercent == 0))
635 {
636 // no flexible time is available
637 dTotalFlexPercent = 0;
638
639 // get a ratio to scale all fixed time entries by
640 dScaleRatio = totalTime_ / dTotalFixedTime;
641
642#ifdef DEBUG_MODE
643 cout << "DVS::generateTimes FIXED dScaleRatio = " << dScaleRatio << endl;
644#endif
645
646 for (int iIndex = 0; iIndex < iNumSegments; iIndex++)
647 {
648 // scale every FIXED time value
649 if (this->getSegmentTimeType(iIndex) == FIXED)
650 {
651 generatedSegmentTimes_->set
652 (iIndex,
653 this->getSegmentTime(iIndex) * dScaleRatio);
654 }
655 else
656 {
657 // all flex time entries are set to 0
658 generatedSegmentTimes_->set (iIndex, 0);
659 }
660 }
661 }
662 // else if their percentages do not add up to 100% (i.e. 1.0) and they are
663 // trying to use flex time, scale the flex time values
664 else if (dTotalFlexPercent != 1.0 && dTotalFlexPercent != 0)
665 {
666 dScaleRatio = 1.0 / dTotalFlexPercent;
667
668#ifdef DEBUG_MODE
669 cout << "DVS::generateTimes FLEXIBLE dScaleRatio = " << dScaleRatio << endl;
670#endif
671
672 // scale percentages
673 for (int iIndex = 0; iIndex < iNumSegments; iIndex++)
674 {
675 if (this->getSegmentTimeType(iIndex) == FLEXIBLE)
676 {
677 generatedSegmentTimes_->set
678 (iIndex,
679 this->getSegmentTime(iIndex) * dScaleRatio);
680 }
681 }
682 }
683
684 // if we have FLEXIBLE time to set, the values stored right now are only
685 // ratios of time available, so we need to set them to actual times
686 if (dTotalFlexPercent > 0)
687 {
688 // calculate the time available to use for flex durations
689 double flexTimeAvailable = totalTime_ - dTotalFixedTime;
690
691#ifdef DEBUG_MODE
692 cout << "DVS::generateTimes flexTimeAvailable = " << flexTimeAvailable << endl;
693#endif
694
695 // set actual durations for FLEXIBLE time segments
696 for (int iIndex = 0; iIndex < iNumSegments; iIndex++)
697 {
698 // if this is a flex-time entry, set time accordingly
699 if (this->getSegmentTimeType(iIndex) == FLEXIBLE)
700 {
701 generatedSegmentTimes_->set
702 (iIndex,
703 generatedSegmentTimes_->get(iIndex) * flexTimeAvailable);
704 }
705 }
706 }
707
708#ifdef DEBUG_MODE
709 cout << "DVS::generateTimes all done. Segments times are:" << endl;
710
711 for (int iIndex = 0; iIndex < iNumSegments; iIndex++)
712 {
713 cout << "Segment " << iIndex << ": " << generatedSegmentTimes_->get(iIndex) << endl;
714 }
715#endif
716}

719//----------------------------------------------------------------------------//
720void DynamicVariableSequence::addInterpolators (m_rate_type rate)
721{
722#ifdef DEBUG_MODE
723 cout << "DVS::addInterpolators" << endl;
724 cout << "num segments: " << segments_->size() << endl;
725 cout << "num points: " << xyPoints_->size() << endl;
726#endif
727
728 int iNumSegments = segments_->size();
729
730 // clear out interpolators that are stored, if there are any stored
731 if (interpolators_ != NULL)
732 {
733 delete (interpolators_);
734 }
735
736 // initialize a new Collection for interpolators
737 interpolators_ = new Collection<Interpolator*> ();
738
739 // for each segment that we have, store a new interpolator
740 for (int iIndex = 0; iIndex < iNumSegments; iIndex++)
741 {
742 // add an interpolator for the current segment to the Collection
743 switch (getSegmentInterpolationType(iIndex))
744 {
745 case EXPONENTIAL:
746 interpolators_->add(new ExponentialInterpolator());
747 break;
748
749 case CUBIC_SPLINE:
750 interpolators_->add(new CubicSplineInterpolator());
751 break;
752
753 default:
754 interpolators_->add(new LinearInterpolator());
755 break;
756 }
757
758 // set the Interpolator's sampling rate to the rate specified
759 interpolators_->get(iIndex)->setSamplingRate(rate);
760
761 m_time_type tempTimeValue = generatedSegmentTimes_->get(iIndex);
762
763 // add a start and end point to this interpolator:
764 interpolators_->get(iIndex)->addEntry(0, xyPoints_->get(iIndex).y);
765 interpolators_->get(iIndex)->addEntry
766 (tempTimeValue,
767 xyPoints_->get(iIndex + 1).y);
768
769 // set interpolator duration
770 interpolators_->get(iIndex)->setDuration(tempTimeValue);
771
772#ifdef DEBUG_MODE
773 cout << "Interpolator #" << iIndex << " - added points: " << endl;
774 cout << " (0, " << xyPoints_->get(iIndex).y << ")" << endl;
775 cout << " (" << generatedSegmentTimes_->get(iIndex) << ", ";
776 cout << xyPoints_->get(iIndex + 1).y << ")" << endl;
777#endif
778 }
779}

782//----------------------------------------------------------------------------//
783void DynamicVariableSequence::scale(m_value_type factor)
784{
785 int iNumPoints = xyPoints_->size();
786 xy_point pointTemp;
787
788 // for every point that we've got stored
789 for (int iIndex = 0; iIndex < iNumPoints; iIndex++)
790 {
791 // scale this value by the factor given
792 pointTemp.x = xyPoints_->get(iIndex).x;

794 xyPoints_->set(iIndex, pointTemp);
795 }
