51Sound::Sound(int numPartials, m_value_type baseFreq)
52{
53
54 spatializer_ = new Spatializer();
55 spa_modified_ = false; /* ZIYUAN CHEN, July 2023 */
56 if (numPartials < 1)
57 {
58 cerr << "ERROR: Sound: Creation with less than 1 partial." << endl;
59 return;
60 }
61
62 // generate each subsequent partial
63 for (int i=0; i<numPartials; i++)
64 {
65 Partial p;
66 p.setParam(RELATIVE_AMPLITUDE, (1/pow(2.71828,i)));
67
68 // INCREMENT FREQUENCY MULTIPLIER FOR GLISSANDO
69 p.setParam(FREQUENCY, baseFreq * (i+1));
70 p.setParam(PARTIAL_NUM, i);
71 add(p);
72 }
73
74 setParam(DURATION, 1.0);
75 setParam(START_TIME, 0.0);
76 setParam(LOUDNESS, 44100.0);
77 setParam(LOUDNESS_RATE, 44100.0);
78
79 setParam(DETUNE_SPREAD, 0.0);
80 setParam(DETUNE_DIRECTION, -1.0); //-1.0
81 setParam(DETUNE_VELOCITY, 0.5);
82 setParam(DETUNE_FUNDAMENTAL, 1.0); //0.0
83
84 reverbObj = NULL;
85
86}

127void Sound::setDetune(double direction, double spread, double velocity){
128
129 if ( !(direction != -1 || direction != 1 )){
130 cerr << "ERROR: Sound: out of range for DETUNE_DIRECTION." << endl;
131 return;
132 }
133
134 if (spread < 0 or spread > 1){
135 cerr << "ERROR: Sound: out of range DETUNE_SPREAD.Should be a percent" << endl;

137 }
138

140 cerr << "ERROR: Sound: out of range for DETUNE_VELOCITY" << endl;

142 }
143
144 if (spread == 0 and direction == 0 and velocity == 0){
145 return;
146 }
147
148 setParam(DETUNE_DIRECTION,direction);
149 setParam(DETUNE_SPREAD,spread);
150 setParam(DETUNE_VELOCITY, velocity);
151// setParam(DETUNE_FUNDAMENTAL, 1);

183//----------------------------------------------------------------------------//
184MultiTrack* Sound::render(
185 int numChannels,
186 m_rate_type samplingRate)
187 // Remember, the default value for samplingRate (as defined in the
188 // .h file is DEFAULT_SAMPLING_RATE
189{
190 //------------------
191 // calculate loudness for this sound.
192 //------------------
193
194 // negative loudness values signify that
195 // loudness is not to be calculated for this sound.
196 if (getParam(LOUDNESS) >= 0)
197 {
198 cout << "\t Calculating Loudness..." << endl;
199 //m_rate_type loudnessRate = m_rate_type(getParam(LOUDNESS_RATE));
200 //Loudness::calculate(*this, loudnessRate);
201 Loudness::calculate(*this);
202 }
203
204 //------------------
205 // do the detuning setup
206 //------------------
207 if (size() != 0) {
208 cout << "\t Creating Envelopes..." << endl;
209 Iterator<Partial> iter = iterator();
210
211 if(getParam(DETUNE_FUNDAMENTAL) == 1.0){
212 cout << "\t using DETUNE" << endl;
213// showDetune();
214
215 while(iter.hasNext()){
216 // create the detuning envelope for this partial
217
218 if (getParam( DETUNE_VELOCITY) == 0.5){
219 LinearInterpolator dv;
220 setup_detuning_env(&dv);
221 iter.next().setParam(DETUNING_ENV,dv);
222 } else{
223 ExponentialInterpolator detuning_env;
224 setup_detuning_env(&detuning_env);
225 iter.next().setParam(DETUNING_ENV,detuning_env);
226 }
227 }
228 }
229 }
230
231 //------------------
232 // render each partial, and composite into a single track.
233 //------------------
234
235 cout << "\t Rendering..." << endl;
236
237 /*
238 * duration gets tricky when you've got partials and sounds, either or both
239 * of which can have different reverb applied. 'duration' refers to the
240 * 'dry' duration of this sound. This is analogous to the length of time an
241 * instrument is buzzing a noise. This is the duration of time we are
242 * generating sine waves in the partial. After 'duration', each partial may
243 * or may not have reverb applied to it, and the overall sound may have
244 * reverb (potentially on top -- meaning, in addition to -- the partials'
245 * reverb.
246 *
247 * in contrast, getTotalDuration() returns the total audible length of time
248 * during which this sound, its partials, and any reverb will produce noise.
249 */
250 m_time_type duration = getParam(DURATION);
251
252 // compute # of samples, taking into account space for reverb die-out
253 m_sample_count_type sampleCount;
254 sampleCount = (m_sample_count_type) ((m_time_type)getTotalDuration() * (m_time_type)samplingRate);
255
256 /* ZIYUAN CHEN, July 2023: Partial::render() now returns (potentially spatialized) MultiTracks */
257 MultiTrack* composite;
258
259 if (size() == 0)
260 {
261 // there are no partials
262 // create a new empty MultiTrack:
263 composite = new MultiTrack(numChannels, sampleCount, samplingRate);
264 // should we zero this memory out?
265 }
266 else
267 {
268 Iterator<Partial> iter = iterator();
269 composite = iter.next().render(numChannels, sampleCount, duration, samplingRate);
270
271 MultiTrack* tempTrack;
272 while(iter.hasNext())
273 {
274 tempTrack = iter.next().render(numChannels, sampleCount, duration, samplingRate);
275 composite->composite(*tempTrack);
276 delete tempTrack;
277 }
278 }
279
280 /* Chain of Conversion:
281 * Track ---do_reverb_Track--> Track in Partial::render()
282 * ... ---spatialize_Track--> MultiTrack in Partial::render()
283 * ... ---do_filter_MultiTrack--> MultiTrack in Sound::render()
284 * ... ---do_reverb_MultiTrack--> MultiTrack in Sound::render()
285 * ... ---spatialize_MultiTrack--> MultiTrack in Sound::render()
286 */
287
288 // do the filter
289 if (filterObj != NULL){
290 cout << "\t Applying Filter..." << endl;
291 MultiTrack &filteredTrack = filterObj->do_filter_MultiTrack(*composite);
292 delete composite;
293 composite = &filteredTrack;
294 }
295
296
297 // do the reverb
298 if (reverbObj != NULL) {
299 cout << "\t Applying Reverb..." << endl;
300 MultiTrack &reverbedTrack = reverbObj->do_reverb_MultiTrack(*composite);
301 delete composite;
302
303 //------------------
304 // spatialize the sound into a MultiTrack object
305 //------------------
306
307 /* ZIYUAN CHEN, July 2023: On the default behavior of spatialization (Sound side)
308 * If a non-placeholder subclass of Spatializer (Pan/MultiPan) is set in the partials,
309 * "spatializer_" in the sound will be a placeholder and should be IGNORED.
310 * Otherwise, (indirectly) calling Spatializer::spatialize_Track() will cause each
311 * track to be averaged across all channels (default placeholder behavior) and
312 * OVERWRITE any spatialization performed by the partials!
313 * Compare Partial::render().
314 */
315 cout << "\t Spatializing..." << endl;
316
317 if (!spa_modified_)
318 return &reverbedTrack;
319
320 MultiTrack* mt = spatializer_->spatialize_MultiTrack(reverbedTrack, numChannels, sampleCount, samplingRate);
321
322 // delete the temporary track object that held the unspatialized reverbed sound
323 delete &reverbedTrack;
324
325 return mt;
326 }
327 else
328 {
329 //------------------
330 // spatialize the sound into a MultiTrack object
331 //------------------
332
333 /* ZIYUAN CHEN, July 2023 - See above */
334 cout << "\t Spatializing..." << endl;
335
336 if (!spa_modified_)
337 return composite;
338
339 MultiTrack* mt = spatializer_->spatialize_MultiTrack(*composite, numChannels, sampleCount, samplingRate);
340 delete composite;
341 return mt;
342 }
343}

371//----------------------------------------------------------------------------//
372void Sound::setup_detuning_env(ExponentialInterpolator *detuning_env)
373{
374//cout << " EXPONENTIAL Interpolator" << endl;
375
376 // it does no detune/tune, return
377 if(getParam(DETUNE_FUNDAMENTAL) != 1.0) {
378 return;
379 }
380
381 double x[3], y[3], spread0, spread, vel;
382 double randy = (float)random() / (float)RAND_MAX;
383
384 // determine the shape of the envelope
385 vel = getParam(DETUNE_VELOCITY);
386 x[0] = 0.0;
387 y[0] = 1.0;
388
389 x[1] = vel * 0.95 + randy * 0.05;
390 y[1] = x[1];
391//cout << "orig x1= " <<x[1] << endl;
392 x[2] = 1.0;
393 y[2] = 0.0;
394
395 // scale by the height spread of the envelope
396 spread0 = getParam(DETUNE_SPREAD);
397 spread = spread0 * randy * 2.0 - spread0;
398/*
399cout << " randy=" << randy << endl;
400cout << "Final spread= " << spread << endl;
401*/
402 y[0] *= spread;
403 y[1] *= spread * (randy * 0.5);
404
405 // then offset to normalize the whole thing at 1.0
406 y[0] += 1.0;
407 y[1] += 1.0;
408 y[2] += 1.0;
409
410 if(getParam(DETUNE_DIRECTION) == -1.0) // divergence (detuning)
411 {
412//cout << " diverging (detuning)" << endl;
413 detuning_env->addEntry(x[0], y[2]);
414 detuning_env->addEntry(x[1], y[1]);
415 detuning_env->addEntry(x[2], y[0]);
416/*
417cout << " x0=" << x[0] << " y2=" << y[2] << endl;
418cout << " x1=" << x[1] << " y1=" << y[1] << endl;
419cout << " x2=" << x[2] << " y0=" << y[0] << endl;
420//int sever; cin >> sever;
421*/
422 } else if(getParam(DETUNE_DIRECTION) == 1.0) { // convergence (tuning)
423//cout << " converging (tuning)" << endl;
424 detuning_env->addEntry(x[0], y[0]);
425 detuning_env->addEntry(x[1], y[1]);
426 detuning_env->addEntry(x[2], y[2]);
427/*
428cout << " x0=" << x[0] << " y0=" << y[0] << endl;
429cout << " x1=" << x[1] << " y1=" << y[1] << endl;
430cout << " x2=" << x[2] << " y2=" << y[2] << endl;
431//int sever; cin >> sever;
432*/
433 }

435}

501//----------------------------------------------------------------------------//
502m_time_type Sound::getTotalDuration(void)
503{
504 m_time_type curDuration, maxDuration;
505 Iterator<Partial> iter = iterator();
506
507 maxDuration = getParam(DURATION);
508 while(iter.hasNext())
509 {
510 curDuration = iter.next().getTotalDuration(getParam(DURATION));
511 if(curDuration > maxDuration)
512 maxDuration = curDuration;
513 }
514
515 maxDuration += ( (reverbObj != NULL) ? reverbObj->getDecay() : 0.0);
516
517 return maxDuration;
518}

522{
523 xmlOutput << "\t<sound>" << endl;
524
525 // Output reverb ID and update reverb collection if necessary

562void Sound::xml_read(XmlReader::xmltag* soundtag, DISSCO_HASHMAP<long, Reverb *>* reverbHash, DISSCO_HASHMAP<long, DynamicVariable *>* dvHash)
563{
564 if(strcmp("sound",soundtag->name))
565 {
566 printf("Not a sound tag?! This is a %s tag!\n", soundtag->name);
567 return;
568 }
569
570 char *value;
571
572 if((value=soundtag->findChildParamValue("reverb_ptr","id")) != 0)
573 if(Reverb* temp = (*reverbHash)[atoi(value)])
574 use_reverb(temp);
575 if((value = soundtag->findChildParamValue("duration","value")) != 0)
576 setParam(DURATION, atof(value));
577 if((value = soundtag->findChildParamValue("start_time","value")) != 0)
578 setParam(START_TIME, atof(value));
579 if((value = soundtag->findChildParamValue("loudness","value")) != 0)
580 setParam(LOUDNESS, atof(value));
581 if((value = soundtag->findChildParamValue("loudness_rate","value")) != 0)
582 setParam(LOUDNESS_RATE, atof(value));
583 if((value = soundtag->findChildParamValue("detune_spread","value")) != 0)
584 setParam(DETUNE_SPREAD, atof(value));
585 if((value = soundtag->findChildParamValue("detune_direction","value")) != 0)
586 setParam(DETUNE_DIRECTION, atof(value));
587 if((value = soundtag->findChildParamValue("detune_velocity","value")) != 0)
588 setParam(DETUNE_VELOCITY, atof(value));
589 if((value = soundtag->findChildParamValue("detune_fundamental","value")) != 0)
590 setParam(DETUNE_FUNDAMENTAL, atof(value));
591
592 XmlReader::xmltag *partialtag;
593
594 while((partialtag=soundtag->children->find("partial")) != 0)
595 {
596 Partial p;
597 p.xml_read(partialtag, reverbHash, dvHash);
598 add(p);
599 }
