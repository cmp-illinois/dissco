68{

70 {
71 // delete any tracks:

99void MultiTrack::composite(MultiTrack& mt, m_time_type startTime)
100{
101
102 Iterator<Track*> it1 = iterator();
103 Iterator<Track*> it2 = mt.iterator();
104
105 while (it1.hasNext() && it2.hasNext())
106 {
107 it1.next()->composite(*it2.next(), startTime);
108 }
109
110 // we should do some warning if sizes aren't equal and such.
