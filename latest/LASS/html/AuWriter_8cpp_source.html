70 va_list marker;
71
72 bool ret_val = true;
73 char *cur_filename = filename;
74 Iterator<Track*> it = mt.iterator();
75
76 va_start(marker, filename);
77 while (it.hasNext())
78 {
79 channels.push_back(& it.next()->getWave());
80 ret_val = ret_val & write(channels, string(cur_filename));
81 channels.pop_back();
82 cur_filename = va_arg(marker, char *);
83 }
84 va_end(marker);

89//----------------------------------------------------------------------------//
90bool AuWriter::write(vector<SoundSample*>& channels, string filename,
91 int bits)
92{
93 // ensure that all of the SoundSamples, have the same samplingRate.
94 // Also, find the lowest sampleCount value of all the SoundSamples.
95 // warn if sections of other SoundSamples are going to be clipped
96 // because of this.
97
98 //Clear out the info structure.
99
100
101 // unused parameter
102 (void) bits;;
103
104 SF_INFO s_info;
105 memset(&s_info, 0, sizeof(SF_INFO));
106
107 //Ensure we actually have channels of sound to work with.
108 if (channels.size() == 0) {
109 cout << "ERROR: AuWriter: numChannels = 0" << endl;
110 return false;
111 }
112
113 //Set the info parameters.
114 s_info.channels = channels.size();
115 s_info.samplerate = channels[0]->getSamplingRate();
116 s_info.format = SF_FORMAT_PCM_24;
117 bits = 0; //Do not use the incoming format, 24-bit is all-purpose.
118
119 //Determine what format to use based on the filename extension.
120 if(filename.find(".wav") != string::npos) {
121 s_info.format = s_info.format | SF_FORMAT_WAV;
122 } else if(filename.find(".aiff") != string::npos) {
123 s_info.format = s_info.format | SF_FORMAT_AIFF;
124 } else if(filename.find(".au") != string::npos) {
125 s_info.format = s_info.format | SF_FORMAT_AU;
126 } else {
127 cout << "ERROR: The output file '" << filename <<
128 "' does not have an extension that is recognizable. Try"
129 " using .wav, .aiff, or .au." << endl;
130 return false;
131 }
132
133 //Open the sound file.
134 SNDFILE* s = sf_open(filename.c_str(), SFM_WRITE, &s_info);
135 if(!s) {
136 cout << "ERROR: The output file '" << filename <<
137 "' could not be opened by libsndfile." << endl;
138 return false;
139 }
140
141 //Determine max number of samples in the channels.
142 m_sample_count_type minSamples = channels[0]->getSampleCount();
143 m_sample_count_type maxSamples = minSamples;
144 for (int c = 0; c < (int)channels.size(); c++) {
145 //Make sure every channel has the same sampling rate.
146 if (channels[c]->getSamplingRate() != channels[0]->getSamplingRate())
147 {
148 cerr << "ERROR: AuWriter: not all channels "
149 << "have the same sampling rate" << endl;
150 return false;
151 }
152
153 //Check the number of samples.
154 m_sample_count_type count = channels[c]->getSampleCount();
155 if (count < minSamples) minSamples = count;
156 if (count > maxSamples) maxSamples = count;
157 }
158
159 //Warn about different amounts of samples.
160 if(maxSamples > minSamples) {
161 m_time_type secondsClipped =
162 ((m_time_type)(maxSamples - minSamples)) /
163 ((m_time_type)channels[0]->getSamplingRate());
164
165 cerr << "WARNING: AuWriter: "
166 << "Because not all SoundSamples were of the same length, " << endl
167 << secondsClipped << " seconds will be clipped off of the "
168 << "end of one or more channels." << endl;
169 }
170
171 //Create multi-channel chunk.
172 int chunkFrames = 1024 * 16;
173 float* chunk = new float[chunkFrames * channels.size()];
174
175
176 int outOfBounds = 0;
177
178 for(m_sample_count_type currentIn = 0; currentIn < minSamples; currentIn += chunkFrames){
179 m_sample_count_type framesToWrite = chunkFrames;
180 if(framesToWrite > minSamples - currentIn)
181 framesToWrite = minSamples - currentIn;
182 for(int c = 0; c < (int)channels.size(); c++) {
183 for(m_sample_count_type i = currentIn; i < currentIn + framesToWrite; i++) {
184 m_sample_type sample = (*channels[c])[i];
185
186 //Check bounds.
187 if (sample > 1.0) {sample = 1.0; outOfBounds++;}
188 if (sample < -1.0) {sample = -1.0; outOfBounds++;}
189
190 chunk[(i - currentIn) * channels.size() + c] = sample;
191 }
192 }
193 sf_writef_float(s, chunk, (sf_count_t)framesToWrite);
194 }
195 sf_close(s);
196
197 delete[] chunk; //added by ming-ching to prevent memory leak. Dec.10 2012
198 return true;
199}
