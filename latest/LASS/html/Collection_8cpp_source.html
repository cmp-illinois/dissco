59template<class T> Collection<T>& Collection<T>::operator=(const Collection<T>& c)
60{
61 if (this != &c) // beware self assignment
62 {
63 // clear this vector
64 clear();
65
66 // copy every element of c into this instance
67 typename vector<T>::iterator it = vector_.begin();
68
69 while(it != vector_.end())
70 {
71 add(*it);
72 it++;
73 }
74 }
75
76 return *this;
77}

95{
96 if ( (index >= 0) && (index < (int) vector_.size()))
97 {
98 // get an iterator pointing at the index
