63void Score::add(Sound* _sound){
64
65 // To prevent unnecessary memory use, this function is temporarily blocked
66 // when there are too many sounds objects waiting to be rendered.
67 // 200 is just an arbitrary number.
68 // This funcion can be upgraded to block the main thread in a fancier way, but
69 // for now this is enough.
70 // --Ming-ching May 06, 2013
71
72 // Rubin Du July 2024: Integrated the resource management, producer thread / consumer thread synchronization, and optimized concurrency by implementing semaphores
73
74 semEmptySlotsSounds.wait(); // Wait for an empty slot in sounds vector
75
76 {
77 std::lock_guard<std::mutex> lock(mutexSoundVector);
78 sounds.push_back(_sound);
79 soundObjectsCreated ++;
80
81 //update scoreEndTime
82 m_time_type soundEndTime = _sound->getParam(START_TIME) +
83 _sound->getTotalDuration();
84 if (soundEndTime > scoreEndTime) {
85 scoreEndTime = soundEndTime;
86 //cout<<"Score End Time Updated: "<< scoreEndTime << "seconds"<<endl;
87 }
88 // figure in the reverb die-out period
89 if(reverbObj != NULL)
90 scoreEndTime += reverbObj->getDecay();
91 }
92
93 semFullSlotsSounds.post(); // Signal that there is a full slot in sounds vector
94
95}

103void render(ThreadEntry* _threadEntry){
104 // Take ownership of the heap-allocated entry passed from the main thread.
105 std::unique_ptr<ThreadEntry> owned(_threadEntry);
106 ThreadEntry& threadEntry = *owned;
107 while(true){
108 threadEntry.semFullSlotsSounds->wait(); // Wait for a full slot in sounds vector
109
110 //Lock the sounds vector
111 threadEntry.mutexSoundVector->lock();
112
113 if (threadEntry.sounds->size()) {
114 *(threadEntry.soundsRendered) = *(threadEntry.soundsRendered) + 1;
115 cout<<"Thread #"<<threadEntry.threadID<<": Rendering sound #"<<
116 *(threadEntry.soundsRendered)<<" of "<<
117 *(threadEntry.soundObjectsCreated)<<endl;
118
119 Sound* sound = threadEntry.sounds->back();
120 threadEntry.sounds->pop_back();
121 threadEntry.mutexSoundVector->unlock();
122
123 // Render the sound outside the critical section
124 MultiTrack* renderedSound = sound->render(threadEntry.numChannels,threadEntry.samplingRate);
125
126 threadEntry.score->addRenderedSound(sound->getParam(START_TIME), renderedSound);
127
128 delete sound;
129
130 threadEntry.semEmptySlotsSounds->post(); // Signal that there is an empty slot in sounds vector
131 }
132 else {
133 threadEntry.mutexSoundVector->unlock();
134 threadEntry.semFullSlotsSounds->post(); // Put back the full slot since no sound was consumed
135 if (threadEntry.score->isDoneGettingSoundObjects()) // If the main thread is done adding sounds, return
136 return;
137 }
138 }// end of main while loop
139
140}

148Score::Score(int _numThreads, int _numChannels, int _samplingRate )
149 : cmm_(NONE),
150 soundsRendered(0),
151 soundObjectsCreated(0),
152 doneGettingSoundObjects(false),
153 workerThreadsAllJoined(false),
154 semEmptySlotsSounds(MAX_SOUND_OBJECTS),
155 semFullSlotsSounds(0),
156 semEmptySlotsRendered(MAX_RENDERED_OBJECTS),
157 semFullSlotsRendered(0),
158 numChannels(_numChannels),
159 samplingRate(_samplingRate)
160{
161 scoreEndTime = 1; //start with a small number
162 scoreMultiTrackLength = scoreEndTime;
163 m_sample_count_type newNumSamples =
164 (m_sample_count_type) (scoreMultiTrackLength * float(samplingRate));
165 scoreMultiTrack = new MultiTrack
166 (numChannels,newNumSamples,samplingRate);
167
168
169 reverbObj = NULL;
170 numThreads = _numThreads;
171
172 //create threads for sound rendering
173 threads = new std::thread[_numThreads];
174
175 for (int i = 0; i < _numThreads; i ++){
176 ThreadEntry* threadEntry = new ThreadEntry;
177 threadEntry->score = this;
178 threadEntry->threadID = i;
179 threadEntry->sounds = &sounds;
180 threadEntry->numChannels = _numChannels;
181 threadEntry->samplingRate = _samplingRate;
182 threadEntry->mutexSoundVector=&mutexSoundVector;
183 threadEntry->soundsRendered = &soundsRendered;
184 threadEntry->soundObjectsCreated = &soundObjectsCreated;
185 threadEntry->scoreMultiTrack =&scoreMultiTrack;
186 threadEntry->semEmptySlotsSounds = &semEmptySlotsSounds;
187 threadEntry->semFullSlotsSounds = &semFullSlotsSounds;
188 threadEntry->semEmptySlotsRendered = &semEmptySlotsRendered;
189 threadEntry->semFullSlotsRendered = &semFullSlotsRendered;
190 threads[i] = std::thread(render, threadEntry);
191 }
192
193 //start the compositeThread
194 compositeThread = std::thread(composite, this);
195
196}

200
201 std::pair<m_time_type, MultiTrack*>* newPair = new pair<m_time_type, MultiTrack*>(_startTime, _renderedSound);

217void Score::compositeRenderedSounds(){
218 while (true){
219 semFullSlotsRendered.wait();
220
221 mutexVectorRenderedSound.lock();
222
223 if(renderedSounds.size()){
224 pair<m_time_type,MultiTrack*>* thisPair = renderedSounds.back();
225 renderedSounds.pop_back();
226 mutexVectorRenderedSound.unlock();
227 semEmptySlotsRendered.post();
228 checkScoreMultiTrackLength();
229 scoreMultiTrack->composite(*(thisPair->second), thisPair->first);
230 delete thisPair->second;
231 delete thisPair;
232 }
233 else{
234 mutexVectorRenderedSound.unlock();
235 semFullSlotsRendered.post();
236 if (workerThreadsAllJoined)
237 return;
238 }
239
240 }

246MultiTrack* Score::joinThreadsAndMix(){
247 //Join the threads
248 for (int i = 0; i < numThreads; i ++){
249 threads[i].join();
250 cout<< "thread Joined: Thread #"<< i <<endl;
251 }
252 cout<<"=======================Threads all joined.==================="<<endl;
253 delete [] threads;
254 workerThreadsAllJoined = true;
255 semFullSlotsRendered.post(); //unblock the composite thread for proper termination
256
257 //join the composite thread
258 compositeThread.join();
259
260 // do the reverb
261 if(reverbObj != NULL)
262 {
263 cout << "Applying reverb to the score..." << endl;
264 MultiTrack *tmp = & reverbObj->do_reverb_MultiTrack(*scoreMultiTrack);
265 delete scoreMultiTrack;
266 scoreMultiTrack = tmp;
267 }
268
269 // perform Clipping management on the composite:
270 cout << "Managing Clipping for the score..." << endl;
271 manageClipping(scoreMultiTrack, cmm_);
272 // return the composite:
273 return scoreMultiTrack;
274}

345void Score::clip(MultiTrack* mt)
346{
347 cout << "Performing CLIP" << endl;
348
349 // for each track
350 for (int t=0; t<mt->size(); t++)
351 {
352 SoundSample& wave = mt->get(t)->getWave();
353 SoundSample& amp = mt->get(t)->getAmp();
354
355 // for each sample
356 m_sample_count_type numSamples = wave.getSampleCount();
357 for (m_sample_count_type s=0; s<numSamples; s++)
358 {
359 if (wave[s] > 1.0) wave[s] = 1.0;
360 if (wave[s] < -1.0) wave[s] = -1.0;
361 if (amp[s] > 1.0) amp[s] = 1.0;
362 if (amp[s] < -1.0) amp[s] = -1.0;
363 }
364
365 }
366}

371void Score::scale(MultiTrack* mt)
372{
373 cout << "Performing SCALE" << endl;
374
375 // -----
376 // first, find the maximum amplitude:
377
378 m_sample_type maxAmp = 0;
379 // for each track
380 for (int t=0; t<mt->size(); t++)
381 {
382 SoundSample& amp = mt->get(t)->getAmp();
383 // for each sample
384 m_sample_count_type numSamples = amp.getSampleCount();
385 for (m_sample_count_type s=0; s<numSamples; s++)
386 {
387 if (amp[s] > maxAmp) maxAmp = amp[s];
388 }
389 }
390
391 // -----
392 // create a scaling factor:
393 m_sample_type scalingFactor = 1.0 / maxAmp;
394
395 // -----
396 // scale every value by this factor
397
398 // for each track
399 for (int t=0; t<mt->size(); t++)
400 {
401 SoundSample& wave = mt->get(t)->getWave();
402 SoundSample& amp = mt->get(t)->getAmp();
403
404 // for each sample
405 m_sample_count_type numSamples = wave.getSampleCount();
406 for (m_sample_count_type s=0; s<numSamples; s++)
407 {
408 wave[s] *= scalingFactor;
409 amp[s] *= scalingFactor;
410 }
411 }
412}

457void Score::anticlip(MultiTrack* mt)
458{
459 cout << "Performing ANTICLIP" << endl;
460
461 // this is a dificult one, because we need fast cros-track access.
462 // make our own data-structure for this one.
463 vector<SoundSample*> wave;
464 vector<SoundSample*> amp;
465 int numTracks = mt->size();
466 for (int i=0; i<numTracks; i++)
467 {
468 wave.push_back(& mt->get(i)->getWave());
469 amp.push_back(& mt->get(i)->getAmp());
470 }
471
472 // now, for each sample:
473 m_sample_count_type numSamples = wave[0]->getSampleCount();
474 for (m_sample_count_type s=0; s<numSamples; s++)
475 {
476 // find the total amplitude across all tracks.
477 m_sample_type totalAmp = 0.0;
478 for (int t=0; t<numTracks; t++)
479 {
480 totalAmp += (*amp[t])[s];
481 }
482
483 // scale if necessary
484 if (totalAmp > 1.0)
485 {
486 m_sample_type scalingFactor = 1.0 / totalAmp;
487 for (int t=0; t<numTracks; t++)
488 {
489 (*amp[t] )[s] *= scalingFactor;
490 (*wave[t])[s] *= scalingFactor;
491 }
492 }
493 }
494}
