202//----------------------------------------------------------------------------//
203m_value_type Envelope::getValue(m_value_type x, m_value_type totalLength)
204{
205 if (x > totalLength || x < 0) {
206 // invalid values, so let them know
207 cout << "Envelope::getValue(...): invalid x or totalLength parameter" << endl;
208 }
209
210 //Handle special case where absolute end of envelope is asked for
211 if (x == totalLength) {
212 return segments_->get(segments_->size() - 1).y;
213 }
214
215 //TODO: Make come sort of cached flag that will see if this
216 // actually needs to be performed or not.
217 generateLengths(totalLength);
218
219 //First, find the segment index that this x/totalLength refers to
220 m_value_type current = 0;
221 int x_Index = 0;
222 while (current < x) {
223 current += generatedSegmentLengths_->get(x_Index);
224 x_Index++;
225 }
226
227 //If we have overshot, lets go back
228 if (current != x) {
229 x_Index--;
230 current -= generatedSegmentLengths_->get(x_Index);
231 }
232
233 //Spawn an interpolator of the proper type
234 Interpolator *interp;
235 switch (getSegmentInterpolationType(x_Index)) {
236 case EXPONENTIAL:
237 interp = new ExponentialInterpolator();
238 break;
239
240 case CUBIC_SPLINE:
241 interp = new CubicSplineInterpolator();
242 break;
243
244 default:
245 interp = new LinearInterpolator();
246 break;
247 }
248
249 // set the Interpolator's sampling rate to a fixed granularity
250 interp->setSamplingRate(100);
251
252 // set interpolator duration to one second to make it easy
253 interp->setDuration(1.0);
254
255 // We know that this segment interpolates between xy point
256 // x_Index and x_Index+1
257 xy_point left, right;
258 left = getPoint(x_Index);
259 right = getPoint(x_Index + 1);
260
261 // Now tell the interpolator what values we are going between
262 interp->addEntry(0, left.y);
263 interp->addEntry(1, right.y);
264
265 // Figure out which sample we want from the interpolator.
266 // This is done by taking the percentage we have to iterate thru
267 // the interpolator, and then multiplying that by the number of samples
268 int sample = (int) (round(((x - current) / generatedSegmentLengths_->get(x_Index)) * 100.0));
269
270 // Create a value iterator to get values from the interpolator
271 Iterator < m_value_type > tempIterator = interp->valueIterator();
272
273 // Now lets skip the first sample-1 samples...
274 for (int i = 0; i < sample; i++) {
275 tempIterator.next();
276 }
277
278 // Now grab the one we want
279 m_value_type returnval = tempIterator.next();
280
281 delete interp;
282
283 return returnval;
284}

360//----------------------------------------------------------------------------//
361void Envelope::addToShape(Collection<envelope_segment> segs)
362{
363 int startIndex; // want to start at 0 if segments_ is empty; otherwise 1
364 m_value_type x_Offset;
365 if (segments_->size() == 0) {
366 x_Offset = 0;
367 startIndex = 0;
368 } else {
369 x_Offset = segments_->get(segments_->size() - 1).x;
370 startIndex = 1;
371
372 // verify that the last old point and the first new point have the
373 // same y value (usually zero), that they can be appended
374 if ((segments_->get(segments_->size() - 1).y) != (segs.get(0).y))
375 {
376 cout << "WARNING: Y values of last point of first envelope\n";
377 cout << "and first point of second envelope do not match.\n";
378 cout << "This implementation corrupts the first point of the\n";
379 cout << "second envelope\n";
380 }
381 }
382 m_value_type x_Last = x_Offset;
383 envelope_segment segTemp;
384
385
386 // for every element in the new collections
387 for (int i = startIndex; i < segs.size(); i++)
388 {
389 // add the current segment and point + 1 to our existing Collections
390 segTemp = segs.get(i);
391 segTemp.x = segTemp.x + x_Offset;
392 segTemp.length = segTemp.x - x_Last;
393 x_Last = segTemp.x;
394 segments_->add(segTemp);
395 }
396}

412Envelope* Envelope::multiply(Envelope & env1, Envelope & env2)
413{
414 // new envelope that will be returned based on this:
415 Collection<envelope_segment> result;
416
417 // X and Y segments
418 Collection<envelope_segment>* env1Segs = env1.getSegments();
419 Collection<envelope_segment>* env2Segs = env2.getSegments();
420
421 envelope_segment seg1, seg2, newseg;
422 m_value_type length=0, max1=0, max2=0;
423 // find the biggest x values
424 for (int i = 0; i < env1Segs->size(); i++) {
425 if (env1Segs->get(i).x > max1) {
426 max1 = env1Segs->get(i).x;
427 }
428 }
429 for (int i = 0; i < env2Segs->size(); i++) {
430 if (env2Segs->get(i).x > max2) {
431 max2 = env2Segs->get(i).x;
432 }
433 }
434 if (max1 >= max2) {
435 length = max1;
436 } else {
437 length = max2;
438 }
439
440 // multiply - find y values for every x value in either envelope
441 Iterator<envelope_segment> segments1 = env1Segs->iterator();
442 Iterator<envelope_segment> segments2 = env2Segs->iterator();
443
444 envelope_segment *add_seg, *first_seg, *second_seg;
445
446 first_seg = &seg1;
447 second_seg = &seg2;
448 add_seg = NULL;
449
450 while (first_seg != NULL || second_seg != NULL) {
451 // advance pointers depending on which seg was added last
452 if (add_seg == &seg1) {
453 printf("was 1\n");
454 if (segments1.hasNext())
455 seg1 = segments1.next();
456 else
457 first_seg = NULL;
458 } else if (add_seg == &seg2) {
459 printf("was 2\n");
460 if (segments2.hasNext())
461 seg2 = segments2.next();
462 else
463 second_seg = NULL;
464 } else { // this is the first time
465 seg1 = segments1.next();
466 seg2 = segments2.next();
467 printf("first time\n");
468 }
469
470 m_value_type seg_len;
471 // determine which point to add now
472 if ((((seg1.x / max1) * length) < ((seg2.x / max2) * length)
473 && first_seg != NULL) || second_seg == NULL) {
474 add_seg = &seg1;
475 seg_len = max1;
476 } else {
477 add_seg = &seg2;
478 seg_len = max2;
479 }
480
481 // MULTIPLICATION
482 // right now you get all FLEXABLE segs, this is good
483 // right now all EXPONENTIAL segs, this is BOGUS :-)
484
485 newseg.x = (add_seg->x / seg_len) * length;
486 newseg.y = env1.getValue(newseg.x, length) *
487 env2.getValue(newseg.x, length);
488 printf("x=%f, length=%f\n", newseg.x, length);
489 newseg.lengthType = FLEXIBLE;
490 newseg.interType = EXPONENTIAL;
491
492 // check for duplicate of x value
493 int duplicate = 0;
494 Iterator <envelope_segment> res_it = result.iterator();
495
496 while (res_it.hasNext()) {
497 envelope_segment oldseg = res_it.next();
498 if (newseg.x < (oldseg.x + ENV_MULT_EPSILON) && newseg.x > (oldseg.x - ENV_MULT_EPSILON)) {
499 duplicate = 1;
500 break;
501 }
502 }
503
504 printf("doing x=%f", newseg.x);
505 // add only if not a duplicate
506 if (duplicate == 0) {
507 result.add(newseg);
508 printf(" added.");
509 }
510 printf("\n");
511 if (first_seg == NULL)
512 printf("first_seg=null\n");
513 if (second_seg == NULL)
514 printf("second_seg=null\n");
515 } // while
516
517 return (new Envelope(result));
518}

674//----------------------------------------------------------------------------//
675Iterator<m_value_type> Envelope::valueIterator()
676{
677 // optimize so that if the envelope is used repeatedly without changing
678 // the time value that is assigned, interpolators are not re-created
679 // every time
680 if (totalLength_ != getDuration() || currentInterpolatorRate_ != getSamplingRate()) {
681
682 totalLength_ = getDuration();
683 currentInterpolatorRate_ = getSamplingRate();
684 // populate the lengths member variable
685 generateLengths(totalLength_);
686
687 // populate the interpolator member variable with interpolators, based
688 // on the lengths just set
689 addInterpolators(currentInterpolatorRate_);
690 }
691
692 // give them the big iterator
693 return (Iterator<m_value_type> (new EnvelopeIterator(*interpolators_)));
694}

696//----------------------------------------------------------------------------//
697void Envelope::generateLengths(m_time_type totalLength)
698{
699 int iNumSegments = segments_->size() - 1;
700 double dTotalFixedLength = 0;
701 double dTotalFlexPercent = 0;
702 double dScaleRatio = 1;
703
704 totalLength_ = totalLength;
705
706 // get rid of previously generated lengths
707 generatedSegmentLengths_->clear();
708
709 // figure out how they want to allocate length for each segment and populate
710 // the generated lengths with their current values
711 for (int iIndex = 0; iIndex < iNumSegments; iIndex++) {
712 if (this->getSegmentLengthType(iIndex) == FIXED) {
713 // if this segment has fixed-length, increment fixed length running total
714 dTotalFixedLength += getSegmentLength(iIndex);
715 } else {
716 // else increment flex-length running total
717 dTotalFlexPercent += getSegmentLength(iIndex);
718 }
719 // append their current value to our values for generated lengths
720 generatedSegmentLengths_->add(getSegmentLength(iIndex));
721 }
722
723 // now, correct the generated values as necessary (i.e. the rest of this)
724
725 // if they specified more FIXED length than actual length, or if there is no
726 // FIEXIBLE length and they did not specify enough FIXED length, scale the
727 // FIXED lengths to the actual length (and set total amount of FLEXIBLE length
728 // to zero)
729 if (dTotalFixedLength > totalLength_
730 || (dTotalFixedLength < totalLength_ && dTotalFlexPercent == 0)) {
731 if (dTotalFixedLength > totalLength_) {
732 cout << "WARNING: Duration specified was less than added FIXED lengths\n"
733 << "FIXED lengths will also be scaled down\n";
734 } else {
735 cout << "WARNING: Duration specified was more than added FIXED lengths\n"
736 << "Since no lengths are FLEXIBLE, FIXED lengths will be scaled up\n";
737 }
738 // no flexible length is available
739 dTotalFlexPercent = 0;
740
741 // get a ratio to scale all fixed length entries by
742 dScaleRatio = totalLength_ / dTotalFixedLength;
743
744 for (int iIndex = 0; iIndex < iNumSegments; iIndex++) {
745 // scale every FIXED length value
746 if (this->getSegmentLengthType(iIndex) == FIXED) {
747 generatedSegmentLengths_->set(iIndex, getSegmentLength(iIndex) * dScaleRatio);
748 setSegmentLength(iIndex, getSegmentLength(iIndex) * dScaleRatio);
749 } else {
750 // all flex length entries are set to 0
751 generatedSegmentLengths_->set(iIndex, 0);
752 setSegmentLength(iIndex, 0);
753 }
754 }
755 } else if (dTotalFlexPercent != 1.0 && dTotalFlexPercent != 0) {
756 // else if their percentages do not add up to 100% (i.e. 1.0) and they are
757 // trying to use flex length, scale the flex length values
758 dScaleRatio = 1.0 / dTotalFlexPercent;
759
760 // scale percentages
761 for (int iIndex = 0; iIndex < iNumSegments; iIndex++) {
762 if (getSegmentLengthType(iIndex) == FLEXIBLE) {
763 generatedSegmentLengths_->set(iIndex, getSegmentLength(iIndex) * dScaleRatio);
764 }
765 }
766 }
767
768 // if we have FLEXIBLE length to set, the values stored right now are only
769 // ratios of length available, so we need to set them to actual lengths
770 if (dTotalFlexPercent > 0) {
771 // calculate the length available to use for flex durations
772 double flexLengthAvailable = totalLength_ - dTotalFixedLength;
773
774 // set actual durations for FLEXIBLE length segments
775 for (int iIndex = 0; iIndex < iNumSegments; iIndex++) {
776 // if this is a flex-length entry, set length accordingly
777 if (getSegmentLengthType(iIndex) == FLEXIBLE) {
778 generatedSegmentLengths_->set(iIndex, generatedSegmentLengths_->get(iIndex) * flexLengthAvailable);
779 }
780 }
781 }
782}

784//----------------------------------------------------------------------------//
785void Envelope::addInterpolators(m_rate_type rate)
786{
787 int iNumSegments = segments_->size();
788 // clear out interpolators that are stored, if there are any stored
789 for (int i = 0; i < interpolators_->size(); i++) {
790 delete(interpolators_->get(i));
791 }
792 interpolators_->clear();
793
794 // initialize a new Collection for interpolators
795 m_value_type prev = segments_->get(0).y;
796
797
798 // for each segment that we have, store a new interpolator
799 for (int iIndex = 0; iIndex < iNumSegments - 1; iIndex++) {
800 // add an interpolator for the current segment to the Collection
801 switch (getSegmentInterpolationType(iIndex)) {
802 case EXPONENTIAL:
803 interpolators_->add(new ExponentialInterpolator());
804 break;
805
806 case CUBIC_SPLINE:
807 interpolators_->add(new CubicSplineInterpolator());
808 break;
809
810 default:
811 interpolators_->add(new LinearInterpolator());
812 break;
813 }
814
815 // set the Interpolator's sampling rate to the rate specified
816 interpolators_->get(iIndex)->setSamplingRate(rate);
817 m_value_type tempLengthValue = generatedSegmentLengths_->get(iIndex);
818
819 // add a start and end point to this interpolator:
820 interpolators_->get(iIndex)->addEntry(0, prev);
821 interpolators_->get(iIndex)->addEntry(tempLengthValue, segments_->get(iIndex + 1).y);
822
823 // set interpolator duration
824 interpolators_->get(iIndex)->setDuration(tempLengthValue);
825 prev = segments_->get(iIndex + 1).y;
826 }
827}
