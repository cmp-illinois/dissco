207void Reverb::ConstructorCommon(Envelope *percentReverb, float *combGainList,
208 float *lpGainList, float gainAllPass, float delay,
209 m_rate_type samplingRate)
210{
211 long combdelay[REVERB_NUM_COMB_FILTERS];
212 int i;
213 Envelope* temp = new Envelope(*percentReverb);
214 delete percentReverb;
215 percentReverb = temp;
216 // figure out allpass filter parameters
217 //gainReverb = percentReverb; //these two lines need fixing
218 // actually, the above and below two lines probably won't be needed
219 //gainDirect = 1 - percentReverb;
220
221 combdelay[0] = (long)(samplingRate*0.050);
222 combdelay[1] = (long)(samplingRate*0.056);
223 combdelay[2] = (long)(samplingRate*0.061);
224 combdelay[3] = (long)(samplingRate*0.068);
225 combdelay[4] = (long)(samplingRate*0.072);
226 combdelay[5] = (long)(samplingRate*0.078);
227
228 // Make comb filters (figure out low-pass gains better)
229 for(i=0;i<REVERB_NUM_COMB_FILTERS;i++)
230 {
231 lpcfilter[i] = new LPCombFilter(combGainList[i], combdelay[i], lpGainList[i]);
232 }
233
234 // Make an all pass filter
235 allPassDelay = delay;
236 apfilter = new AllPassFilter(gainAllPass, (long)((m_time_type)allPassDelay*samplingRate) );
237
238 // Find the decay time
239 float alpha = 0.0; // alpha is the steady state gain (which is also the max of the gain fn
240#define max(x,y) ((x) > (y) ? (x) : (y))
241 for(i=0;i<6;i++)
242 alpha = max(alpha, combGainList[i]/(1.0 - lpGainList[i]));
243 float T_r = -3.0 * delay / log(alpha);
244 decay_duration = T_r*1.0;
245}

269m_sample_type Reverb::do_reverb(m_sample_type x_t, float x_value, Envelope *percentReverbinput)
270{
271 m_sample_type y;
272 Envelope* temp = new Envelope(*percentReverbinput);
273 delete percentReverb;
274 percentReverb = temp;
275
276 // run the sample through various comb filters (for effeciency
277 // reasons, I hard coded this (instead of looping from 0 to
278 // (REVERB_NUM_COMB_FILTERS-1).
279 y = lpcfilter[0]->do_filter(x_t);
280 y += lpcfilter[1]->do_filter(x_t);
281 y += lpcfilter[2]->do_filter(x_t);
282 y += lpcfilter[3]->do_filter(x_t);
283 y += lpcfilter[4]->do_filter(x_t);
284 y += lpcfilter[5]->do_filter(x_t);
285 y /= (m_sample_type)REVERB_NUM_COMB_FILTERS;
286
287 // after adding up the results, run it through an allpass filter
288 y = apfilter->do_filter(y);
289 // Mix it with the input sound
290 float durationofEnv = percentReverb->getDuration();
291 float EnvelopeValueAtx = percentReverb->getValue(x_value,durationofEnv);
292 y = (EnvelopeValueAtx*y) + ((1 - EnvelopeValueAtx)*x_t);
293 return y;
294}

322MultiTrack &Reverb::do_reverb_MultiTrack(MultiTrack &inWave, Envelope *percentReverbinput)
323{
324 MultiTrack *newMultiTrack;
325 SoundSample *newWave, *newAmp;
326
327 // create the new MultiTrack
328 newMultiTrack = new MultiTrack();
329
330 Envelope* temp = new Envelope(*percentReverbinput);
331 delete percentReverb;
332 percentReverb = temp;
333 // iterate over all individual tracks
334 // for each track:
335 // grab the SoundSample from inside the Track
336 // filter the SoundSample, which returns a NEW SoundSample
337 // reset the filter so that it can be used fresh next time
338 // create a new track based on the returned, filtered SoundSample
339 // add this new track to the output MultiTrack
340 Iterator<Track*> it = inWave.iterator();
341 while (it.hasNext())
342 {
343 Track *curTrack = it.next();
344
345 newWave = do_reverb_SoundSample(&curTrack->getWave(), percentReverb);
346 reset();
347 newAmp = constructAmp(newWave);
348 reset();
349 newMultiTrack->add(new Track(newWave, newAmp));
350 }
351
352 return *newMultiTrack;

390SoundSample *Reverb::do_reverb_SoundSample(SoundSample *inWave, Envelope *percentReverbinput)
391{
392 SoundSample *outWave;
393
394 Envelope* temp = new Envelope(*percentReverbinput);
395 delete percentReverb;
396 percentReverb = temp;
397
398 outWave = new SoundSample(inWave->getSampleCount(),
399 inWave->getSamplingRate());
400
401#ifdef __APPLE__
402 // macOS buffer path: process entire audio buffer at once using
403 // do_filter_buffer (tight loops / vDSP SIMD) instead of per-sample calls.
404 long N = (long)inWave->getSampleCount();
405 float* inData = &(*inWave)[0];
406 float* outData = &(*outWave)[0];
407
408 // Accumulate the 6 comb filter outputs.
409 vector<float> combSum(N, 0.0f);
410 vector<float> filterBuf(N);
411 for (int f = 0; f < REVERB_NUM_COMB_FILTERS; f++) {
412 lpcfilter[f]->do_filter_buffer(inData, filterBuf.data(), N);
413 vDSP_vadd(combSum.data(), 1, filterBuf.data(), 1,
414 combSum.data(), 1, (vDSP_Length)N);
415 }
416
417 // Scale by 1 / REVERB_NUM_COMB_FILTERS.
418 float scale = 1.0f / (float)REVERB_NUM_COMB_FILTERS;
419 vDSP_vsmul(combSum.data(), 1, &scale, combSum.data(), 1, (vDSP_Length)N);
420
421 // All-pass filter into outData.
422 apfilter->do_filter_buffer(combSum.data(), outData, N);
423
424 // Precompute envelope values and apply wet/dry mix:
425 // out[i] = env[i] * apOut[i] + (1-env[i]) * in[i]
426 // = env[i] * (apOut[i] - in[i]) + in[i]
427 float duration = percentReverb->getDuration();
428 float invN = 1.0f / (float)N;
429 vector<float> envVals(N);
430 for (long i = 0; i < N; i++)
431 envVals[i] = percentReverb->getValue((float)i * invN, duration);
432
433 // diff[i] = apOut[i] - in[i] (vDSP_vsub: C = B - A)
434 vector<float> diff(N);
435 vDSP_vsub(inData, 1, outData, 1, diff.data(), 1, (vDSP_Length)N);
436
437 // outData[i] = envVals[i] * diff[i] + in[i]
438 vDSP_vma(envVals.data(), 1, diff.data(), 1, inData, 1,
439 outData, 1, (vDSP_Length)N);
440
441#else
442 for (int i = 0; i < inWave->getSampleCount(); i++)
443 (*outWave)[i] = do_reverb((*inWave)[i],
444 (float)i / inWave->getSampleCount(),
445 percentReverb);
446#endif
447
448 return outWave;
449}

497SoundSample *Reverb::constructAmp(SoundSample *wave)
498{
499 m_sample_count_type nSamples, windowRadius;
500 long ctr1, pos, maxAmpPos;
501 m_rate_type sampRate;
502 SoundSample *ampWave;
503 m_sample_type maxAmp;
504
505 // initialize various variables
506 // loop over all samples
507 nSamples = wave->getSampleCount();
508 sampRate = wave->getSamplingRate();
509 ampWave = new SoundSample(nSamples, sampRate, false);
510 windowRadius = (m_sample_count_type) (1.0/20.0*sampRate / 2.0);
511
512 // find the initial max (over a window of plus/minus windowRadius,
513 // with the center at index=0 of the wave SoundSample
514 maxAmp = (*wave)[0];
515 maxAmpPos = 0;
516 for(ctr1=1; ctr1<min(windowRadius,nSamples); ctr1++)
517 {
518 if((*wave)[ctr1] > maxAmp)
519 {
520 maxAmp = (*wave)[ctr1];
521 maxAmpPos = ctr1;
522 }
523 }
524
525 // loop over all samples
526 for(; ctr1<nSamples; ctr1++)
527 {
528 // has the old maximum gone out of range (too far to the left
529 // of the window) ?
530 if(maxAmpPos <= (ctr1 - windowRadius))
531 {
532 // yes, so find a new maximum over the whole window
533 maxAmp = 0.0;
534 for(pos = max(ctr1-windowRadius,0); pos < min(ctr1+windowRadius,nSamples); pos++)
535 {
536 if((*wave)[pos] > maxAmp)
537 {
538 maxAmp = (*wave)[pos];
539 maxAmpPos = pos;
540 }
541 }
542 }
543 else if(maxAmp < (*wave)[min(ctr1+windowRadius, nSamples)])
544 {
545 maxAmp = (*wave)[min(ctr1+windowRadius, nSamples)];
546 maxAmpPos = min(ctr1+windowRadius, nSamples);
547 }
548 (*ampWave)[ctr1] = maxAmp;
549 }
550 return ampWave;
551}

577void Reverb::xml_read(XmlReader::xmltag *reverbtag)
578{
579
580 // Sanity Check
581 if(strcmp("reverb",reverbtag->name))
582 {
583 printf("Not a reverb tag! This is a %s tag!\n",reverbtag->name);
584 return;
585 }
586
587 char *value;
588 if((value = reverbtag->findChildParamValue("gainDirect","value")) != 0)
589 set_gainDirect(atof(value));
590
591 if((value = reverbtag->findChildParamValue("gainReverb","value")) != 0)
592 set_gainReverb(atof(value));
593
594 if((value = reverbtag->findChildParamValue("allPassDelay","value")) != 0)
595 set_allPassDelay(atof(value));
596
597 if((value = reverbtag->findChildParamValue("decay_duration","value")) != 0)
598 set_decay_duration(atof(value));
599
600 XmlReader::xmltag *childtag;
601 int lpIndex = 0;
602 while((childtag = reverbtag->children->find("LPCombFilter")) != 0)
603 {
604 LPCombFilter* lp = new LPCombFilter();
605 lp->xml_read(childtag);
606 setLPComb(lpIndex, lp);
607 lpIndex++;
608 }
609
610 while((childtag = reverbtag->children->find("AllPassFilter")) != 0)
611 {
612 AllPassFilter* apf = new AllPassFilter();
613 apf->xml_read(childtag);
614 setAllPass(apf);
615 }
616}
