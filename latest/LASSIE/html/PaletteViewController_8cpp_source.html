20// Sorts children within folders alphabetically, but keeps folders in insertion order.
21class PaletteSortProxy : public QSortFilterProxyModel {
22public:
23 using QSortFilterProxyModel::QSortFilterProxyModel;
24
25protected:
26 bool lessThan(const QModelIndex &left, const QModelIndex &right) const override {
27 // Root-level items (folders): preserve insertion order
28 if (!left.parent().isValid())
29 return left.row() < right.row();
30 // Children: sort alphabetically by display text
31 return QSortFilterProxyModel::lessThan(left, right);
32 }

314 if (!item || item->column() != 1) {

316 }
317
318 // Get the parent item to determine the folder/event type
319 QStandardItem* parent = item->parent();
320 if (!parent) {
321 return; // This is a folder name, not an event name
322 }
323
324 // Get the new name from the item
325 QString newName = item->text();
326
327 // Get the row index within the parent folder
328 int index = item->row();
329
330 // Get the event type from the type column (column 0)
331 QStandardItem* typeItem = parent->child(item->row(), 0);

340 // Get project manager

342
343 // Retrieve old name from backend and reject the rename if newName is a duplicate.
344 QString oldName;
345 {
346 bool isDuplicate = false;

348 auto checkHList = [&](QList<HEvent>& list) {
349 if (index >= list.size()) return;
350 oldName = list[index].name;
351 for (int i = 0; i < list.size(); ++i)
352 if (i != index && list[i].name == newName) isDuplicate = true;
353 };
354 auto checkSList = [&](const auto& list) {
355 if (index >= list.size()) return;
356 oldName = list[index].name;
357 for (int i = 0; i < list.size(); ++i)
358 if (i != index && list[i].name == newName) isDuplicate = true;
359 };
360
361 if (eventType == "High") checkHList(pm->highevents());
362 else if (eventType == "Mid") checkHList(pm->midevents());
363 else if (eventType == "Low") checkHList(pm->lowevents());
364 else if (eventType == "Bottom") {
365 const auto& list = pm->bottomevents();
366 if (index < list.size()) {
367 oldName = list[index].event.name;
368 for (int i = 0; i < list.size(); ++i)
369 if (i != index && list[i].event.name == newName) isDuplicate = true;
370 }
371 }
372 else if (eventType == "Spectrum") checkSList(pm->spectrumevents());
373 else if (eventType == "Note") checkSList(pm->noteevents());
374 else if (eventType == "Envelope") checkSList(pm->envelopeevents());
375 else if (eventType == "Sieve") checkSList(pm->sieveevents());
376 else if (eventType == "Spatialization") checkSList(pm->spaevents());
377 else if (eventType == "Pattern") checkSList(pm->patternevents());
