113inline QString eventtypeToDisplayString(int type) {
114 switch (type) {
115 case top: return "Top";
116 case high: return "High";
117 case mid: return "Mid";
118 case low: return "Low";
119 case bottom: return "Bottom";
120 case sound: return "Spectrum";
121 case env: return "Envelope";
122 case sieve: return "Sieve";
123 case spa: return "Spatialization";
124 case pattern: return "Pattern";
125 case reverb: return "Reverb";
126 case folder: return "Folder";
127 case note: return "Note";
128 case filter: return "Filter";
129 default: return QString::number(type);
130 }
131}
