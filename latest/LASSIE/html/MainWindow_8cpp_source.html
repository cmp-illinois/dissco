52static QString resolveCmodBinary()
53{
54 // Prefer a CMOD sitting next to the LASSIE binary (packaged builds:
55 // LASSIE.app/Contents/MacOS/CMOD on macOS, the AppDir's usr/bin/CMOD on
56 // Linux). Fall back to the compile-time CMOD_BINARY path for dev builds.
57 const QString appDir = QCoreApplication::applicationDirPath();
58 for (const QString &candidate : { appDir + "/CMOD", appDir + "/../Resources/CMOD" }) {
59 if (QFileInfo(candidate).isExecutable()) {
60 return QDir::cleanPath(candidate);
61 }
62 }
63 return QStringLiteral(CMOD_BINARY);
64}

624void MainWindow::refreshRecentProjects() const
625{
626 const QSettings settings;
627 const QStringList recents = settings.value(kRecentProjectsKey).toStringList();
628
629 ui->recentProjectsList->clear();
630 ui->recentProjectsTree->clear();
631
632 const QFileIconProvider iconProvider;
633 const QIcon fallbackIcon = iconProvider.icon(QFileIconProvider::File);
634 const QLocale locale;
635
636 for (const QString &path : recents) {
637 const QFileInfo info(path);
638
639 QIcon icon = iconProvider.icon(info);
640 if (icon.isNull())
641 icon = fallbackIcon;
642
643 // Icon-mode list
644 auto *listItem = new QListWidgetItem(ui->recentProjectsList);
645 listItem->setText(info.completeBaseName());
646 listItem->setToolTip(QDir::toNativeSeparators(path));
647 listItem->setData(Qt::UserRole, path);
648 listItem->setIcon(icon);
649 listItem->setTextAlignment(Qt::AlignHCenter | Qt::AlignTop);
650
651 // Details tree (list mode)
652 auto *treeItem = new QTreeWidgetItem(ui->recentProjectsTree);
653 treeItem->setText(0, info.completeBaseName());
654 treeItem->setIcon(0, icon);
655 treeItem->setToolTip(0, QDir::toNativeSeparators(path));
656 treeItem->setData(0, Qt::UserRole, path);
657
658 if (info.exists()) {
659 treeItem->setText(1, locale.toString(info.lastModified(), QLocale::ShortFormat));
660 treeItem->setText(2, locale.formattedDataSize(info.size()));
661 } else {
662 treeItem->setText(1, tr("(missing)"));
663 treeItem->setText(2, QString());
664 }
665 }

668void MainWindow::addToRecentProjects(const QString &path) const
669{
670 if (path.isEmpty())
671 return;
672
673 const QString canonical = QFileInfo(path).absoluteFilePath();
674
675 QSettings settings;
676 QStringList recents = settings.value(kRecentProjectsKey).toStringList();
677
678 // Remove any existing entry that refers to the same file, then prepend.
679 recents.removeAll(canonical);
680 recents.removeAll(path);
681 recents.prepend(canonical);
682 while (recents.size() > kMaxRecentProjects)
683 recents.removeLast();
684
685 settings.setValue(kRecentProjectsKey, recents);
686 refreshRecentProjects();
687}
