105static void writeDomNode(QXmlStreamWriter& xmlWriter, const QDomNode& node) {
106 if (node.isElement()) {
107 QDomElement element = node.toElement();
108 xmlWriter.writeStartElement(element.tagName());
109 QDomNamedNodeMap attrs = element.attributes();
110 for (int i = 0; i < attrs.count(); ++i) {
111 QDomAttr attr = attrs.item(i).toAttr();
112 xmlWriter.writeAttribute(attr.name(), attr.value());
113 }
114 for (QDomNode child = element.firstChild(); !child.isNull();
115 child = child.nextSibling()) {
116 writeDomNode(xmlWriter, child);
117 }
118 xmlWriter.writeEndElement();
119 } else if (node.isCDATASection()) {
120 xmlWriter.writeCDATA(node.toCDATASection().data());
121 } else if (node.isText()) {
122 // Skip whitespace-only text nodes (formatting between elements) so the
123 // output stays compact, but keep real values like "STEREO" or "250".
124 const QString text = node.toText().data();
125 if (!text.trimmed().isEmpty()) {
126 xmlWriter.writeCharacters(text);
127 }
128 }

133 QDomDocument tempDoc;

136 QDomNodeList children = tempDoc.documentElement().childNodes();
137 for (int i = 0; i < children.count(); ++i) {
138 writeDomNode(xmlWriter, children.at(i));
139 }
140 } else {
141 // String/Number
142 xmlWriter.writeCharacters(xmlString);
143 }

146// Returns the modifier field value only if it is active for the modifier's type;
147// returns an empty string for fields disabled by the type, so CMOD ignores them.
148static QString modActiveField(const Modifier& mod, int fieldIdx) {
149 // columns: prob(0), amp(1), rate(2), width(3), spread(4), dir(5), vel(6)
150 static const bool table[7][7] = {
151 /* TREMOLO */ { true, true, true, false, false, false, false },
152 /* VIBRATO */ { true, true, true, false, false, false, false },
153 /* GLISSANDO*/ { true, true, false, false, false, false, false },
154 /* DETUNE */ { true, false, false, false, true, true, true },
155 /* AMPTRANS */ { true, true, true, true, false, false, false },
156 /* FREQTRANS*/ { true, true, true, true, false, false, false },
157 /* WAVE_TYPE*/ { false, true, false, false, false, false, false },
158 };
159 if (mod.type >= 7 || !table[mod.type][fieldIdx]) return {};
160 switch (fieldIdx) {
161 case 0: return mod.probability;
162 case 1: return mod.amplitude;
163 case 2: return mod.rate;
164 case 3: return mod.width;
165 case 4: return mod.detune_spread;
166 case 5: return mod.detune_direction;
167 case 6: return mod.detune_velocity;
168 default: return {};
169 }
170}
